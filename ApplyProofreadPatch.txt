        private void ApplyProofreadToDocument(List<ProofreadItem> items)
        {
            try
            {
                var doc = Globals.ThisAddIn?.Application?.ActiveDocument;
                if (doc == null) return;

                int processed = 0;
                
                // 开启修订模式
                bool oldTrackRevisions = doc.TrackRevisions;
                doc.TrackRevisions = true;
                
                try
                {
                    foreach (var item in items)
                    {
                        Word.Range searchRange = doc.Content;
                        if (searchRange.Find.Execute(FindText: item.Original, MatchCase: false, MatchWholeWord: false,
                                                      MatchWildcards: false, Forward: true, Wrap: Word.WdFindWrap.wdFindStop))
                        {
                            // 添加批注说明
                            string commentText = string.Format(
                                "【校对建议】\n" +
                                "类型: {0}\n" +
                                "原文: {1}\n" +
                                "建议修改: {2}\n" +
                                "理由: {3}",
                                item.Type, item.Original, item.Modified, item.Reason);
                            doc.Comments.Add(searchRange, commentText);
                            
                            // 执行修订：替换文本
                            // 这会显示为删除线（原文）+ 下划线（新文本）
                            searchRange.Text = item.Modified;
                            
                            processed++;
                        }
                    }
                }
                finally
                {
                    // 恢复原来的修订状态
                    doc.TrackRevisions = oldTrackRevisions;
                }

                if (processed > 0)
                {
                    AddMessageBubble("系统", "已将 " + processed + " 条诊断以修订形式写入文档。请在「审阅」选项卡中使用「接受」或「拒绝」按钮处理。", false);
                    
                    // 显示修订标记
                    try
                    {
                        var app = Globals.ThisAddIn?.Application;
                        app?.ActiveWindow?.View?.MarkupMode = Word.WdRevisionsMode.wdMixedRevisions;
                    }
                    catch { }
                }
            }
            catch (Exception ex)
            {
                AddMessageBubble("错误", "写回文档时出错: " + ex.Message, false, true);
            }
        }
